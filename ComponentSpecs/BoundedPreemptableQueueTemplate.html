template <class T, int maxLength>
class BoundedPreemptableQueue1
   //! requires: 1 ≤ maxLength
    
   //! is modeled by string of T
   //! exemplar self
   //! constraint |self| ≤ maxLength
{
public: // Standard Operations
   BoundedPreemptableQueue1 ();
      //! replaces self
      //! ensures: self = <>
   ~BoundedPreemptableQueue1 ();
   void clear (void);
      //! clears self
   void transferFrom (BoundedPreemptableQueue1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   BoundedPreemptableQueue1& operator = (BoundedPreemptableQueue1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // BoundedPreemptableQueue1 Specific Operations
   void enqueue (T& x);
      //! updates self
      //! clears x
      //! requires: |self| + 1 ≤ maxLength
      //! ensures: self = #self * <#x>
   void inject (T& x);
      //! updates self
      //! clears x
      //! requires |self| + 1 ≤ maxLength
      //! ensures: self = <#x> * #self
   void dequeue (T& x);
      //! updates self
      //! replaces x
      //! requires: self /= <>
      //! ensures: <x> is prefix of #self  and  self = #self[1, |#self|)
   void replaceFront (T& x);
      //! updates self, x
      //! requires: self /= <>
      //! ensures: <x> is prefix of #self  and  self = <#x> * #self[1, |#self|)
   T& front (void);
      //! restores self
      //! requires: self /= <>
      //! ensures: <front> is prefix of self
   Integer length (void);
      //! restores self
      //! ensures: length = |self|
   Integer remainingCapacity (void);
      //! restores self
      //! ensures: remainingCapacity = maxLength - |self|
            
private: // Representation
}
