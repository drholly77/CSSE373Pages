    //! replaces self
    //! ensures: self = <>

    ~Mystery2();

    void op1(Mystery2& x);
    //! replaces self
    //! replaces x
    //! ensures: self = #x and x = #self

    Mystery2& op2 (Mystery2& x);
    //! replaces self
    //! restores x
    //! ensures: self = x

    //! restores self

    //! restores self

    //! replaces x
    //! restores z
    //! requires: |self| > z >= 0
    //! ensures: self = #self[0, z) * #self[z+1, |#self|) and <x> = #self[z, z+1)

    void op11(T& x);
    //! updates self, x
    //! requires: 0 < |self|
    //! ensures: <x> is suffix of #self  and  self = #self[1, |#self|) * <#x>

    //! restores self
    //! requires: 0 < |self|
    //! ensures: <op12> is suffix of self
      
private: // Representation
}
