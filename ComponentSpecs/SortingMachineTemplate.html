template <class T, Boolean(*areOrdered)(T&, T&)>
class SortingMachine1 : public StdOps<SortingMachine1<T, areOrdered>>, SortingMachineKernel<T>
   //! math subtype SORTING_MACHINE_MODEL 
   //!   is (contents: finite multiset of T, inserting: boolean)

   //! type SortingMachine1 is modeled by SORTING_MACHINE_MODEL
   //!   exemplar self
   //!   initialization ensures self.contents = {} and self.inserting = true

   //! definition IS_FIRST (s: finite multiset of T, x: T): boolean
   //!   is x ∈ s  and  ∀ y: T (y ∈ s → areOrdered (x, y))
{
public: // Standard Operations
   SortingMachine1 ();
      //! replaces self
      //! ensures: self = self.contents = {} and self.inserting = true
   ~SortingMachine1 ();
   void clear (void);
      //! clears self
   void transferFrom (SortingMachine1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   SortingMachine1& operator = (SortingMachine1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // SortingMachine1 Specific Operations
   void add (T& x);
      //! updates self.contents
      //! clears x
      //! requires: self.inserting
      //! ensures: self = (#self.contents ∪ {#x},#self.inserting)
   void changeToExtractionMode (void);
      //! updates self.inserting
      //! requires: self.inserting
      //! ensures: self = (#self.contents,false)
   void removeFirst (T& x);
      //! updates self.contents
      //! replaces x
      //! requires: not self.inserting  and  |self.contents| > 0
      //! ensures: IS_FIRST (#self.contents, x)  and
      //!          self = (#self.contents \ {x},#self.inserting)
   Integer size (void);
      //! restores self
      //! ensures: size = |self.contents|
   Boolean isInInsertionMode (void);
      //! restores self
      //! ensures: isInInsertionMode = self.inserting
         
private: // Representation
};
