Stack Template

 
template <class T>
class Stack1
   //! is modeled by string of T
   //! exemplar self
{
public: // Standard Operations
   Stack1 ();
      //! replaces self
      //! ensures: self = <>
   ~Stack1 ();
   void clear (void);
      //! clears self
   void transferFrom (Stack1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   Stack1& operator = (Stack1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // Stack1 Specific Operations
   void push (T& x);
      //! updates self
      //! clears x
      //! ensures: self = <#x> * #self
   void pop (T& x);
      //! updates self
      //! replaces x
      //! requires: self /= <>
      //! ensures: <x> is prefix of #self  and  self = #self[1, |#self|)
   void replaceTop (T& x);
      //! updates self, x
      //! requires: self /= <>
      //! ensures: <x> is prefix of #self  and  self = <#x> * #self[1, |#self|)
   T& top (void);
      //! restores self
      //! requires: self /= <>
      //! ensures: <top> is prefix of self
   Integer length (void);
      //! restores self
      //! ensures: length = |self|
      
private: // Representation
};
