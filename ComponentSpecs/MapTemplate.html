Map Template

 
template <class K, class V, class KCallback>
class Map1
   //! is modeled by PARTIAL_FUNCTION
   //! math subtype PARTIAL_FUNCTION is finite set of
   //!                  key: K
   //!                  value: V
   //!             exemplar m
   //!             constraint
   //!                ∀ key: K
   //!                (∀ r1, r2: V
   //!                          (((key, r1) ∈ m and (key, r2) ∈ m) → (r1 = r2)))
   //! 
   //!          math operation DOMAIN(m: PARTIAL_FUNCTION): finite set of K satisfies 
   //!               ∀ key: K (key ∈ DOMAIN(m) ↔ ∃ value: V ((key, value) ∈ m))
   //! exemplar self
{
public: // Standard Operations
   Map1 ();
      //! replaces self
      //! ensures: self = {}
   ~Map1 ();
   void clear (void);
      //! clears self
   void transferFrom (Map1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   Map1& operator = (Map1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // Map1 Specific Operations
   void add (K& key, V& value);
      //! updates self
      //! clears key, value
      //! requires: key ∉ DOMAIN(self)
      //! ensures: self = #self ∪ {(#key, #value)}
   void remove (K& key, K& keyFromMap, V& valueFromMap);
      //! updates self
      //! restores key
      //! replaces keyFromMap, valueFromMap
      //! requires: key ∈ DOMAIN(self)
      //! ensures: (key, valueFromMap) ∈ #self  and
      //!           self = #self \ {(keyFromMap, valueFromMap)}  and  keyFromMap = key
   V& value (K& key);
      //! restores self, key
      //! requires: key ∈ DOMAIN(self)
      //! ensures: (key, value) ∈ self
   void removeAny (K& key, V& value);
      //! updates self
      //! replaces key, value
      //! requires: |self| > 0
      //! ensures: (key, value) ∈ #self  and  self = #self \ {(key, value)}
   Boolean hasKey (K& key);
      //! restores self, key
      //! ensures: hasKey = (key ∈ DOMAIN(self))
   Integer size (void);
      //! restores self
      //! ensures: size = |self|

private:   // Representation
};
