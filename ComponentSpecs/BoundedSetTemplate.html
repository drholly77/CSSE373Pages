BoundedSet Template

 
template <class T, class TCallback, int maxSize>
class BoundedSet1
   //! requires: 1 ≤ maxSize

   //! is modeled by FiniteSet of T
   //! exemplar self
   //! constraint |self| ≤ maxSize
{
public: // Standard Operations
   BoundedSet1 ();
      //! replaces self
      //! ensures: self = {}
   ~BoundedSet1 ();
   void clear (void);
      //! clears self
   void transferFrom (BoundedSet1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   BoundedSet1& operator = (BoundedSet1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // BoundedSet1 Specific Operations
   void add (T& x);
      //! updates self
      //! clears x
      //! requires: x ∉ self  and  |self| + 1 ≤ maxSize
      //! ensures: self = #self ∪ {#x}
   void remove (T& xFromClient, T& xFromSet);
      //! updates self
      //! restores xFromClient
      //! replaces xFromSet
      //! requires: xFromClient ∈ self
      //! ensures: self = #self \ {xFromSet}  and  xFromSet = xFromClient
   void removeAny (T& x);
      //! updates self
      //! replaces x
      //! requires: |self| > 0
      //! ensures: self = #self \ {x}
   Boolean contains (T& x);
      //! restores self, x
      //! ensures: contains = (x ∈ self)
   Integer size (void);
      //! restores self
      //! ensures: size = |self|
   Integer remainingCapacity (void);
      //! restores self
      //! ensures: remainingCapacity = maxSize - |self|
      
private: // Representation
};
