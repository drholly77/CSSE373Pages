Set Template

 
template <class T, class TCallback>
class Set1
   //! is modeled by FiniteSet of T
   //! exemplar self
{
public: // Standard Operations
   Set1 ();
      //! replaces self
      //! ensures: self = {}
   ~Set1 ();
   void clear (void);
      //! clears self
   void transferFrom (Set1& source);
      //! replaces self
      //! clears source
      //! ensures: self = #source
   Set1& operator = (Set1& rhs);
      //! replaces self
      //! restores rhs
      //! ensures: self = rhs

   // Set1 Specific Operations
   void add (T& x);
      //! updates self
      //! clears x
      //! requires: x ∉ self
      //! ensures: self = #self ∪ {#x}
   void remove (T& xFromClient, T& xFromSet);
      //! updates self
      //! restores xFromClient
      //! replaces xFromSet
      //! requires: xFromClient ∈ self
      //! ensures: self = #self \ {xFromSet}  and  xFromSet = xFromClient
   void removeAny (T& x);
      //! updates self
      //! replaces x
      //! requires: |self| > 0
      //! ensures: self = #self \ {x}
   Boolean contains (T& x);
      //! restores self, x
      //! ensures: contains = (x ∈ self)
   Integer size (void);
      //! restores self
      //! ensures: size = |self|
private: // Representation
};
