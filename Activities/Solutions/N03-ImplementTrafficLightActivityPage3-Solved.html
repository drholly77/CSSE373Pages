TrafficLight1::TrafficLight1()
{
   // data member state is type Integer
   // Integer's constructor initializes variables to zero
   // Let zero = (true,false,false)
}	// TrafficLight1

TrafficLight1::~TrafficLight1()
{
   // TODO: nothing
}	// ~TrafficLight1

void TrafficLight1::clear(void)
{
   state.clear();
} // clear

void TrafficLight1::transferFrom(TrafficLight1& source)
{
   state.transferFrom(source.state);
} // transferFrom

TrafficLight1& TrafficLight1::operator = (TrafficLight1& rhs)
{
   state = rhs.state;
   return *this; // Must have this return statement at the end
} // operator =

void TrafficLight1::changeState(void)
{
   state = (state + 1) % 3;
}	// change

void TrafficLight1::getState(Boolean& red, Boolean& yellow, Boolean& green)
{
   switch (state) {
      case 0:
         red = true;
         yellow = false;
         green = false;
         break;
      case 1:
         red = false;
         yellow = true;
         green = false;
         break;
      case 2:
         red = false;
         yellow = false;
         green = true;
         break;
      default:
         break;
   } // end switch
} // getState

wostream& operator<<(wostream& os, TrafficLight1& t)
{
	Boolean red, yellow, green;
	t.getState(red, yellow, green);

	os << "(" << red << "," << yellow << "," << green << ")";
	return os;
}
