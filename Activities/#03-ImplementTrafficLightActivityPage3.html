#include "Wrapper.h"
#include "TrafficLight1.hpp"

//-------------------------------------------------------------------
// main executable body
//-------------------------------------------------------------------

int main(int argc, char* argv[])
{
	Boolean showMenu = (argc == 1);
	TrafficLight1 t1, t2;
	Boolean red, yellow, green;

	wcout << "t1 = " << t1 << endl;
	t1.changeState();
	wcout << "t1 = " << t1 << endl;

	t2.transferFrom(t1);
	wcout << "t1 = " << t1 << endl;
	wcout << "t2 = " << t2 << endl;

	t2.changeState();
	wcout << "t2 = " << t2 << endl;

	t2.changeState();
	t2.getState(red, yellow, green);
	if (red) {
		wcout << "the light is red";
	}
	else if (yellow) {
		wcout << "the light is yellow";
	} else {
		wcout << "the light is green";
	} // end if
	wcout << endl;

	wcout << "end program" << endl;
	return 0;
}	// end main
